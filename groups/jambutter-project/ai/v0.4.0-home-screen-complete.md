```typescript
// Quick start function
const startQuickPractice = (minutes: number) => {
  showMessage({
    text: "Let's go! You got this!",
    mood: "excited",
    voice: true,
  });
  navigate("/practice", { state: { startWithMinutes: minutes } });
};

// Streak calculation
function calculateStreak(practiceHistory: Record<string, number>): number {
  const today = new Date();
  let streak = 0;

  for (let i = 0; i < 365; i++) {
    const date = new Date(today);
    date.setDate(date.getDate() - i);
    const dateKey = date.toDateString();

    if (practiceHistory[dateKey] && practiceHistory[dateKey] > 0) {
      streak++;
    } else if (i > 0) {
      break; // Don't break streak if today hasn't been practiced yet
    }
  }

  return streak;
}
```
